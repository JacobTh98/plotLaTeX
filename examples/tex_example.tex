\section*{Examples}
The following are some examples of applications of the plot\LaTeX package.

\setcounter{section}{1}
\subsection{Single line plot}

\begin{figure}[h]
    \centering
    \tikzstyle{every node}=[font=\footnotesize]
    \begin{tikzpicture}
        \begin{axis}[
            ylabel={Relative error in \%},
			xlabel={Index $k$},
            xtick={0,256,...,1024},
            width=7.5cm,
			height=3cm,
			at={(0cm,0cm)},
			scale only axis,
			axis background/.style={fill=white},
            grid=both,
            legend columns = 3,
            legend style={at={(0,1.05)}, legend cell align=left, align=left, draw=white!15!black, mark options={draw=none}, anchor=south west},
         ]
        \addplot[] 
        	table[x=x,y=mean_err,col sep=comma]{data/test_results.csv};
        \addlegendentry{${\mu}_{\textbf{E}}$};
        \end{axis}
    \end{tikzpicture}
    \caption{Caption of the plot.}
    \label{fig:fig1}
\end{figure}

\subsection{Multiple line plot}
\begin{figure}[h]
    \centering
    \tikzstyle{every node}=[font=\footnotesize]
    \begin{tikzpicture}
        \begin{axis}[
            ylabel={Relative error in \%},
			xlabel={Index $k$},
            xtick={0,256,...,1024},
            width=7.5cm,
			height=3cm,
			at={(0cm,0cm)},
			scale only axis,
			axis background/.style={fill=white},
            grid=both,
            legend columns = 3,
            legend style={at={(0,1.05)}, legend cell align=left, align=left, draw=white!15!black, mark options={draw=none}, anchor=south west},
         ]
        % mean_err,std_err,var_err
        \addplot[color=mycolor1,  line width=1.5pt, solid] 
        	table[x=x,y=mean_err,col sep=comma]{data/test_results.csv};
        \addlegendentry{${\mu}_{\textbf{E}}$}; %Mean
        
        \addplot[color=mycolor2,  line width=1.5pt, solid] 
        	table[x=x,y=std_err,col sep=comma]{data/test_results.csv};
        \addlegendentry{$\sigma_{\textbf{E}}^{2}$};%Standard deviation
        \addplot[color=mycolor3,  line width=1.5pt, solid] 
        	table[x=x,y=var_err,col sep=comma]{data/test_results.csv};
        \addlegendentry{Var($\textbf{E})$}; %Variance
        \end{axis}
    \end{tikzpicture}
    \caption{Caption of the plot.}
    \label{fig:fig2}
\end{figure}
